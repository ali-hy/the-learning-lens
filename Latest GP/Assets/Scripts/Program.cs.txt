using System;
using System.Diagnostics;

class Program
{
    static void Main()
    {
        string pythonExe = "python";

        // Path to your script (make sure spaces are handled)
        string scriptName = "\"C:\\Users\\mahmo\\Desktop\\Code\\XYZ\\python script\\evaluation.py\"";

        // Your arguments
        string[] argsarr = { "Mahmuod", "Is", "A", "Good", "Guy" };

        // Wrap each argument in quotes in case they contain spaces
        string args = string.Join(" ", argsarr.Select(a => $"\"{a}\""));

        ProcessStartInfo psi = new ProcessStartInfo
        {
            FileName = pythonExe,
            Arguments = $"{scriptName} {args}",
            RedirectStandardOutput = true,
            RedirectStandardError = true,
            UseShellExecute = false,
            CreateNoWindow = true
        };

        Process process = new Process { StartInfo = psi };
        process.Start();

        string output = process.StandardOutput.ReadToEnd();
        string error = process.StandardError.ReadToEnd();
        process.WaitForExit();
        if (error != "")
            Console.WriteLine(error);
        Console.WriteLine("Output from python: \n" + output);
    }
}
